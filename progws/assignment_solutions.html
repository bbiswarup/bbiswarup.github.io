library("plotly")

# Program parameters
n <- 10000

# Generating Points
x <- runif(n, 0, 1)
y <- runif(n, 0, 1)

points <- data.frame(x, y)

# Checking location
points$loc <- (points$x ^ 2 + points$y ^2 <= 1)

approx_pi <- 4*length(points$loc[points$loc])/n
points$labels <- points$loc
points$labels[points$loc] = 'inside'
points$labels[!points$loc] = 'outside'

# Plotting points
fig <- plot_ly(points, x=x, y=~y, color=~loc, type='scatter',mode='markers', name=~labels)
approx_pi

library(ggplot2)
x1=seq(from=-pi,to=pi,length.out =100)
y1 <- cos(x1)
mydata=data.frame(x=x1,cos=y1)

ptype <- rep("all",length(x=x1))
ptype[x1<=-pi/2] <- "x<=-pi/2"
ptype[x1>-pi/2 & x1<=pi/2] <- "-pi/2<=x<=pi/2"
ptype[x1>pi/2]<-"x>pi/2"
ptype <- factor(x=ptype)

ggplot(data = mydata)+
  geom_ribbon(aes(x=x, ymax=cos, ymin=0), fill="gray")+
  geom_point(aes(x=x,y = cos, colour =ptype,shape=ptype))+
  xlab("x") +
  ylab("cos(x)")

library(plotly)

n = 20
v = seq(1, 10, length.out=n)

# Generating values

x <- seq(-10, 10, length.out = 100)
y <- ((gamma((v[1]+1)/2))/sqrt(v[1] * pi * gamma(v[1]/2))) * ((1 + (x^2/v[1])) ^ (-(v[1]+1)/2))
plot_data = data.frame(xs=x, ys=y, vi = v[1])

for (i in 2:n){
  y <- ((gamma((v[i]+1)/2))/sqrt(v[i] * pi * gamma(v[i]/2))) * ((1 + (x^2/v[i])) ^ (-(v[i]+1)/2))
  new_row = data.frame(xs=x, ys=y, vi = v[i])
  plot_data = rbind(plot_data, new_row)
}

fig <- plot_data %>%
  plot_ly(
    x = ~xs,
    y = ~ys,
    frame = ~vi,
    type = 'scatter',
    mode = 'lines',
    showlegend = F
  )
fig
